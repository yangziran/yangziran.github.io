Java8 启动脚本
2020-06-06 23:03 folder [Java]   label run script   
#!/bin/bash
#--------------------------------------------
# Java程序启动脚本，启动Jar或者基于SpringBoot的War包
# 基于Java8 HotSpot，Linux|MacOS系统
# 使用方式：sh xxx.sh [ start|stop|restart|status ]
# 配置APP_NAME、WORK_DIR、GC_LOGS之后运行脚本即可
# Author Yang ZiRan
# Email	mail@yangziran.com
# Site https://yangziran.com
#--------------------------------------------

# Jar|War文件名称
APP_NAME=""
# Jar|War工作路径
WORK_DIR=""
# GC日志路径
GC_LOGS="/var/logs"
# 进程停止等待时间，秒
SHUTDOWN_WAIT=10

# 启用核心转储，并限制文件大小为1024K，如需不限制文件大小，则使用：ulimit -c unlimited
ulimit -c 1024

# 模式选项
MODE_OPTS="-server"
# 内存选项
MEM_OPTS="-Xmx4G -Xms4G -XX:MetaspaceSize=512M -XX:MaxMetaspaceSize=512M"
# 启动时预申请内存
MEM_OPTS="$MEM_OPTS -XX:+AlwaysPreTouch"
# GC选项
GC_OPTS="-XX:+UseG1GC -XX:MaxGCPauseMillis=100 -XX:+ParallelRefProcEnabled"
GCLOG_OPTS="-Xloggc:${GC_LOGS}/gc.log -XX:+PrintGCDetails -XX:+PrintGCDateStamps -XX:+PrintCommandLineFlags"
# OOM选项
OOM_OPTS="-verbose:class -XX:HeapDumpPath=${GC_LOGS} -XX:+HeapDumpOnOutOfMemoryError"
# CRASH选项
CRASH_OPTS="-XX:ErrorFile=${GC_LOGS}/hs_err_pid%p.log"
# JMX选项
#JMX_OPTS="-Djava.rmi.server.hostname=127.0.0.1 -Dcom.sun.management.jmxremote.port=9999 -Dcom.sun.management.jmxremote.ssl=false -Dcom.sun.management.jmxremote.authenticate=false"
# 其他选项
OTHER_OPTS="-Djava.net.preferIPv4Stack=true -Dfile.encoding=UTF-8"

# JVM启动参数
JAVA_OPTS="$MODE_OPTS $MEM_OPTS $GC_OPTS $GCLOG_OPTS $OOM_OPTS $CRASH_OPTS $OTHER_OPTS"

# 使用说明，用来提示输入参数
usage() {
	echo "Usage: sh xxx.sh [ start|stop|restart|status ]"
	exit 1
}

# 检查程序是否在运行
checkpid() {
	pid=`ps -ef|grep $APP_NAME|grep -v grep|awk '{print $2}'`
	# 如果不存在返回：1，存在返回：0
	if [ -z "${pid}" ]; then
		return 1
	else
		return 0
	fi
}

# 启动方法
start() {
	checkpid
	if [ $? -eq "0" ]; then
		echo "${APP_NAME} is already running. Pid is ${pid}"
	else
		nohup java $JAVA_OPTS -jar $WORK_DIR/APP_NAME > /dev/null 2>&1 &
		checkpid
		echo "${APP_NAME} started. Pid is ${pid}"
	fi 
}

# 停止方法
stop() {
	checkpid
	if [ $? -eq "0" ]; then
		kill $pid

		local count=0
		until [ `ps -p $pid|grep -c $pid` = '0' ] || [ $count -gt $SHUTDOWN_WAIT ]; do
			echo "waiting for processes to exit"
			sleep 1
			let count=$count+1
		done

		if [ $count -gt $SHUTDOWN_WAIT ]; then
			echo "killing processes didn't stop after $SHUTDOWN_WAIT seconds"
			kill -9 $pid
		fi

		echo "${APP_NAME} stopped"
	else
		echo "${APP_NAME} is not running"
	fi
}

# 重启方法
restart() {
	stop
	start
}

# 运行状态
status() {
	checkpid
	if [ $? -eq "0" ]; then
		echo "${APP_NAME} is running. Pid is ${pid}"
	else
		echo "${APP_NAME} is not running"
	fi
}

# 根据输入参数，选择执行对应方法，默认执行使用说明
case "$1" in
	"start")
		start
	;;
	"stop")
		stop
	;;
	"restart")
		restart
	;;
	"status")
		status
	;;
	*)
		usage
	;;
esac
